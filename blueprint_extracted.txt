SECTION 01
Product Vision & Feature Map
What Khao GPT is, what it is not, and every feature explained

The Core Idea
Khao GPT is a taste-intelligent food discovery assistant for Delhi NCR. Every other platform recommends based on popularity and past order history. Khao GPT recommends based on how your palate actually works — a model called your Taste DNA. It then redirects you directly to Zomato or Swiggy to complete the order. No payments. No delivery. Just the intelligence layer that was missing.

Taste DNA Scoring Algorithm
# backend/services/taste_engine.py
 
from dataclasses import dataclass
 
@dataclass
class TasteProfile:
    heat_ceiling: float
    sweet_tolerance: float
    acid_affinity: float
    umami_affinity: float
    fat_palate: float
    bitter_tolerance: float
    cuisine_scores: dict
 
@dataclass
class Dish:
    heat_level: float
    sweet_level: float
    acid_level: float
    umami_level: float
    fat_level: float
    bitter_level: float
    cuisine_type: str
    texture_tags: list
 
 
def score_dish_for_user(dish: Dish, profile: TasteProfile) -> float:
    """
    Returns 0-100 match score. Higher = better fit for this user's palate.
    Penalties subtract from 100. Bonuses add up to +20.
    """
    score = 100.0
 
    # HEAT — hard ceiling. Exceeding it is a strong penalty.
    heat_over = dish.heat_level - profile.heat_ceiling
    if heat_over > 0:
        score -= heat_over * 9          # -9 pts per unit over ceiling
    elif heat_over < -3:
        score -= abs(heat_over) * 1.5  # slight penalty for too mild if user likes heat
 
    # ACID — reward match, penalise mismatch
    acid_diff = abs(dish.acid_level - profile.acid_affinity)
    score -= acid_diff * 3.5
 
    # SWEET — reward match
    sweet_diff = abs(dish.sweet_level - profile.sweet_tolerance)
    score -= sweet_diff * 3
 
    # FAT / RICHNESS
    fat_diff = abs(dish.fat_level - profile.fat_palate)
    score -= fat_diff * 3
 
    # UMAMI — bonus if user loves it and dish has it
    umami_bonus = (dish.umami_level / 10) * (profile.umami_affinity / 10) * 15
    score += umami_bonus
 
    # BITTER — only penalise if user avoids it and dish has it
    if profile.bitter_tolerance < 4 and dish.bitter_level > 5:
        score -= (dish.bitter_level - 5) * 5
 
    # CUISINE AFFINITY
    cuisine_score = profile.cuisine_scores.get(dish.cuisine_type, 5.0)
    score += (cuisine_score - 5) * 2   # -10 to +10 swing
 
    return round(max(0, min(100, score)), 1)
 
 
def update_taste_dna(
    profile: TasteProfile,
    dish: Dish,
    rating: int               # 1=loved, 2=okay, 3=disliked
) -> TasteProfile:
    """
    Nudges Taste DNA after each meal rating.
    Learning rate is small — profile evolves slowly, not jerkily.
    After 10+ ratings, profile is meaningfully personalised.
    """
    lr = 0.06                          # 6% nudge per rating
    d = 1 if rating == 1 else (-1 if rating == 3 else 0)
 
    def nudge(val, dish_val):
        return max(0.0, min(10.0, val + d * lr * dish_val))
 
    profile.heat_ceiling    = nudge(profile.heat_ceiling,   dish.heat_level)
    profile.acid_affinity   = nudge(profile.acid_affinity,  dish.acid_level)
    profile.fat_palate      = nudge(profile.fat_palate,     dish.fat_level)
    profile.sweet_tolerance = nudge(profile.sweet_tolerance, dish.sweet_level)
    profile.umami_affinity  = nudge(profile.umami_affinity, dish.umami_level)
 
    # Cuisine score update
    current = profile.cuisine_scores.get(dish.cuisine_type, 5.0)
    profile.cuisine_scores[dish.cuisine_type] = round(
        max(0, min(10, current + d * 0.4)), 2
    )
 
    return profile

First Bite Simulator
This turns a dish's taste vector + user's Taste DNA into a personalised paragraph describing how the food will taste. It uses template logic — no LLM required.
# backend/services/first_bite.py
 
def generate_first_bite(dish: Dish, profile: TasteProfile) -> str:
    parts = []
 
    # Opening — texture or dominant flavour
    if dish.texture_tags:
        parts.append(f"First thing you will notice is the {dish.texture_tags[0]} texture.")
 
    # Heat personalisation
    heat_over = dish.heat_level - profile.heat_ceiling
    if heat_over > 2:
        parts.append("This runs hotter than your usual comfort zone — the heat builds and lingers.")
    elif heat_over > 0:
        parts.append("Slightly above your usual range — you will feel it but it should be manageable.")
    elif dish.heat_level < 3:
        parts.append("Very mild on heat — no burn at all.")
    else:
        parts.append(f"Heat sits around your {int(dish.heat_level)}/10 comfort level — right in your zone.")
 
    # Acid
    if dish.acid_level > 6 and profile.acid_affinity > 6:
        parts.append("The tang is strong and you will love that.")
    elif dish.acid_level > 6 and profile.acid_affinity < 4:
        parts.append("There is notable sourness here which may feel sharp for your palate.")
    elif dish.acid_level < 3:
        parts.append("No sourness — clean and rounded flavour.")
 
    # Richness
    if dish.fat_level > 7:
        parts.append("Rich and heavy — one portion is a full meal.")
    elif dish.fat_level < 3:
        parts.append("Light dish — will not leave you feeling heavy.")
 
    # Sweet
    if dish.sweet_level > 6 and profile.sweet_tolerance < 4:
        parts.append("Worth noting — this is on the sweeter side, which you typically avoid.")
 
    return " ".join(parts)

Google Places Seeder Script
Run this once to populate your restaurant database for Delhi NCR. Uses the free Google Places API.
# data_pipeline/scrapers/google_places_seeder.py
 
import httpx
import psycopg2
import time
 
API_KEY = "YOUR_GOOGLE_PLACES_API_KEY"   # free from console.cloud.google.com
DB_URL  = "postgresql://user:pass@localhost/khaogpt"
 
# Delhi NCR areas to seed — start with these, expand later
AREAS = [
    {"name": "Malviya Nagar Delhi",     "lat": 28.5355, "lng": 77.2100},
    {"name": "Hauz Khas Delhi",         "lat": 28.5494, "lng": 77.2001},
    {"name": "Saket Delhi",             "lat": 28.5244, "lng": 77.2090},
    {"name": "Lajpat Nagar Delhi",      "lat": 28.5677, "lng": 77.2434},
    {"name": "Connaught Place Delhi",   "lat": 28.6315, "lng": 77.2167},
    {"name": "Karol Bagh Delhi",        "lat": 28.6519, "lng": 77.1909},
    {"name": "Cyber City Gurugram",     "lat": 28.4959, "lng": 77.0882},
    {"name": "Sector 29 Gurugram",      "lat": 28.4677, "lng": 77.0631},
]
 
def search_restaurants(lat, lng, area_name):
    url = "https://maps.googleapis.com/maps/api/place/nearbysearch/json"
    params = {
        "location": f"{lat},{lng}",
        "radius": 1500,
        "type": "restaurant",
        "key": API_KEY
    }
    resp = httpx.get(url, params=params)
    return resp.json().get("results", [])
 
def insert_restaurant(conn, place, area_name):
    cur = conn.cursor()
    name   = place.get("name", "")
    lat    = place["geometry"]["location"]["lat"]
    lng    = place["geometry"]["location"]["lng"]
    rating = place.get("rating", None)
    types  = place.get("types", [])
 
    # Map Google types to cuisine tags
    cuisine_map = {
        "indian_restaurant": "north_indian",
        "restaurant": "multi_cuisine",
        "cafe": "cafe",
        "bakery": "bakery",
    }
    cuisine_tags = [cuisine_map.get(t, t) for t in types if t in cuisine_map]
 
    cur.execute("""
        INSERT INTO restaurants (name, area, lat, lng, cuisine_tags, rating_zomato, source)
        VALUES (%s, %s, %s, %s, %s, %s, 'google_places')
        ON CONFLICT DO NOTHING
    """, (name, area_name, lat, lng, cuisine_tags, rating))
    conn.commit()
 
def run():
    conn = psycopg2.connect(DB_URL)
    for area in AREAS:
        print(f"Seeding {area['name']}...")
        places = search_restaurants(area["lat"], area["lng"], area["name"])
        for place in places:
            insert_restaurant(conn, place, area["name"])
        time.sleep(0.5)                # be polite to the API
    print(f"Done. Total areas seeded: {len(AREAS)}")
    conn.close()
 
if __name__ == "__main__":
    run()

Taste Tag to Taste Vector Inference
When a user submits a community spot, they describe it in taste tags. We need to convert those tags to the numerical taste vector that the recommender uses.
# backend/services/community_service.py
 
TASTE_TAG_MAP = {
    # Heat
    "spicy":         {"heat_level": 7.5},
    "very spicy":    {"heat_level": 9.0},
    "mild":          {"heat_level": 2.0},
    "no heat":       {"heat_level": 0.0},
    # Acid
    "tangy":         {"acid_level": 7.5},
    "sour":          {"acid_level": 8.0},
    "fresh":         {"acid_level": 5.0},
    # Fat
    "rich":          {"fat_level": 8.0},
    "heavy":         {"fat_level": 8.5},
    "light":         {"fat_level": 2.5},
    "oily":          {"fat_level": 8.0},
    # Sweet
    "sweet":         {"sweet_level": 8.0},
    "not sweet":     {"sweet_level": 1.5},
    # Umami
    "umami":         {"umami_level": 8.0},
    "savoury":       {"umami_level": 7.0},
    "smoky":         {"umami_level": 7.5},
    # Texture
    "crispy":        {"texture_tags": ["crispy"]},
    "crunchy":       {"texture_tags": ["crunchy"]},
    "soft":          {"texture_tags": ["soft"]},
    "chewy":         {"texture_tags": ["chewy"]},
    "street_style":  {},
    "cheap":         {},
    "filling":       {"fat_level": 7.0, "umami_level": 6.0},
}
 
TEXTURE_TAGS = ["crispy", "crunchy", "soft", "chewy", "fluffy", "juicy"]
 
def infer_taste_vector(taste_tags: list) -> dict:
    """
    Converts user's taste tags into a numeric taste vector.
    Averages overlapping values when multiple tags map to same axis.
    """
    axis_values = {
        "heat_level": [], "sweet_level": [], "acid_level": [],
        "umami_level": [], "fat_level": [], "bitter_level": []
    }
    texture = []
 
    for tag in taste_tags:
        mapping = TASTE_TAG_MAP.get(tag.lower().strip(), {})
        for axis, val in mapping.items():
            if axis == "texture_tags":
                texture.extend(val)
            elif axis in axis_values:
                axis_values[axis].append(val)
 
    result = {}
    for axis, vals in axis_values.items():
        result[axis] = round(sum(vals) / len(vals), 1) if vals else 5.0
 
    result["texture_tags"] = texture
    return result

Extracting taste intelligence from thousands of reviews automatically

How Taste Vectors Get Built From Reviews
Each dish's taste vector (heat_level, acid_level etc.) is not entered manually. It is computed by mining hundreds of reviews for that dish using NLP, then averaging the taste signals. This is the core data work that makes the First Bite Simulator accurate.

def extract_taste_signals(review_text: str) -> dict:
    """
    Extracts taste signals from one review.
    Returns dict of {axis: [values]}
    """
    text = review_text.lower()
    signals = defaultdict(list)
 
    for phrase, (axis, direction, value) in TASTE_LEXICON.items():
        if phrase in text:
            if axis == "texture":
                signals["texture"].append(value)
            else:
                signals[axis].append(float(value))
 
    return dict(signals)
 
 
def compute_dish_taste_vector(reviews: list) -> dict:
    """
    Takes list of review strings for a dish.
    Returns averaged taste vector across all reviews.
    """
    all_signals = defaultdict(list)
 
    for review in reviews:
        signals = extract_taste_signals(review)
        for axis, values in signals.items():
            all_signals[axis].extend(values if isinstance(values, list) else [values])
 
    vector = {}
    for axis in ["heat_level","sweet_level","acid_level","umami_level","fat_level"]:
        vals = all_signals.get(axis, [5.0])
        vector[axis] = round(sum(vals) / len(vals), 2)
 
    # Most common textures
    textures = all_signals.get("texture", [])
    from collections import Counter
    vector["texture_tags"] = [t for t, _ in Counter(textures).most_common(3)]
 
    return vector

How Zomato and Swiggy deep links are built and tracked

How Deep Links Work
Both Zomato and Swiggy use public URL structures for their restaurant and dish pages. You build these URLs from your database and redirect the user. You do not need any API access, no agreements, no API keys for this basic flow.

Same page — user scrolls or searches for dish

swiggy.com with item pre-highlighted if item ID known

Order Bridge Service
# backend/services/order_bridge.py
 
import re
from urllib.parse import quote
 
def build_zomato_url(restaurant_name: str, area: str, stored_url: str = None) -> str:
    """
    Returns Zomato order URL.
    Uses stored URL from DB if available (most accurate).
    Falls back to constructing from name + area.
    """
    if stored_url:
        return stored_url
 
    # Construct slug from name: "Moti Mahal Delux" → "moti-mahal-delux"
    slug = re.sub(r'[^a-z0-9]+', '-', restaurant_name.lower()).strip('-')
    area_slug = re.sub(r'[^a-z0-9]+', '-', area.lower()).strip('-')
    return f"https://www.zomato.com/ncr/{slug}-{area_slug}/order"
 
 
def build_swiggy_url(restaurant_name: str, area: str, stored_url: str = None) -> str:
    if stored_url:
        return stored_url
 
    slug = re.sub(r'[^a-z0-9]+', '-', restaurant_name.lower()).strip('-')
    return f"https://www.swiggy.com/restaurants/{slug}-{area}"
 
 
def build_order_links(restaurant, dish=None) -> dict:
    """
    Returns both platform links for a recommendation.
    Also builds a Google Maps link as fallback for community/street spots.
    """
    zomato = build_zomato_url(
        restaurant.name, restaurant.area, restaurant.zomato_url
    )
    swiggy = build_swiggy_url(
        restaurant.name, restaurant.area, restaurant.swiggy_url
    )
    maps = (
        f"https://maps.google.com/?q={quote(restaurant.name + ' ' + restaurant.area)}"
        if restaurant.lat is None else
        f"https://maps.google.com/?q={restaurant.lat},{restaurant.lng}"
    )
 
    return {
        "zomato":      zomato,
        "swiggy":      swiggy,
        "google_maps": maps,          # always available — crucial for street food / community spots
        "is_community": restaurant.is_community,
        "note": "Visit in person" if restaurant.is_community and not restaurant.zomato_url else None
    }

@router.post("/chat")
async def chat(request: ChatRequest, current_user = Depends(get_current_user)):
    """
    The main endpoint. Takes a natural language food request.
    Returns top 3 personalised recommendations.
    """
    # 1. Parse intent from message
    intent = parse_intent(request.message)
    # intent = {
    #   "area": "Malviya Nagar",
    #   "budget_max": 300,
    #   "mood": "spicy",
    #   "include_community": True    # if user says "local" or "street food"
    # }
 
    # 2. Fetch user taste profile
    profile = await get_taste_profile(current_user.id)
 
    # 3. Get candidate restaurants (geo + budget filter)
    candidates = await get_restaurants(
        area=intent.get("area"),
        max_cost=intent.get("budget_max"),
        include_community=intent.get("include_community", True)
    )
 
    # 4. Score all dishes across candidates
    scored_dishes = []
    for restaurant in candidates:
        for dish in restaurant.dishes:
            # True cost check
            true_cost = dish.price + restaurant.avg_delivery_fee + restaurant.platform_fee_est
            if intent.get("budget_max") and true_cost > intent["budget_max"]:
                continue
 
            score = score_dish_for_user(dish, profile)
            first_bite = generate_first_bite(dish, profile)
            links = build_order_links(restaurant, dish)
 
            scored_dishes.append({
                "score":       score,
                "dish":        dish,
                "restaurant":  restaurant,
                "first_bite":  first_bite,
                "true_cost":   true_cost,
                "order_links": links
            })
 
    # 5. Sort by score, return top 3
    top3 = sorted(scored_dishes, key=lambda x: x["score"], reverse=True)[:3]
 
    return {
        "recommendations": [format_recommendation(r) for r in top3],
        "message_id": generate_id()
    }

          <span className="text-xs text-green-700 bg-green-50 px-2 py-1 rounded-full">
            Local spot
          </span>
        )}
      </div>
 
      {/* Dish + restaurant name */}
      <h3 className="text-lg font-bold text-gray-900">{dish.name}</h3>
      <p className="text-sm text-gray-500 mb-3">{restaurant.name} · {restaurant.area}</p>
 
      {/* First Bite description */}
      <div className="bg-amber-50 border-l-4 border-amber-400 px-3 py-2 rounded-r-lg mb-3">
        <p className="text-xs font-semibold text-amber-700 mb-1">First Bite</p>
        <p className="text-sm text-gray-700">{first_bite}</p>
      </div>
 
      {/* True cost */}
      <p className="text-sm text-gray-600 mb-3">
        True cost: <span className="font-semibold text-gray-900">₹{true_cost}</span>
        <span className="text-gray-400 text-xs"> (incl. delivery + fees)</span>
      </p>
 
      {/* Order buttons */}
      <OrderButtons links={order_links} dishId={dish.id} restaurantId={restaurant.id} />
    </div>
  );
}

Community Submit Form
// src/components/Community/SubmitSpot.jsx
 
import { useState } from 'react';
import PhotoUploader from './PhotoUploader';
import axios from 'axios';
 
const SPOT_TYPES = ['Restaurant','Cafe','Dhaba','Street Stall','Home Kitchen','Food Cart'];
const TASTE_TAGS = ['Spicy','Very Spicy','Mild','Tangy','Sweet','Sour','Rich',
                    'Light','Crispy','Soft','Smoky','Oily','Street Style','Cheap & Filling'];
 
export default function SubmitSpot() {
  const [form, setForm] = useState({
    spot_name:'', spot_type:'', area:'', address:'',
    description:'', must_try_dish:'', dish_description:'',
    approx_price:'', taste_tags:[], zomato_url:'', swiggy_url:''
  });
  const [photos, setPhotos]   = useState([]);
  const [status, setStatus]   = useState(null);   // null | 'loading' | 'success' | 'error'
  const [message, setMessage] = useState('');
 
  const toggleTag = (tag) =>
    setForm(f => ({
      ...f,
      taste_tags: f.taste_tags.includes(tag)
        ? f.taste_tags.filter(t => t !== tag)
        : [...f.taste_tags, tag]
    }));
 
  const handleSubmit = async () => {
    if (!form.spot_name || !form.area || !form.must_try_dish || photos.length === 0) {
      setMessage('Please fill all required fields and add at least one photo.');
      return;
    }
    setStatus('loading');
    const data = new FormData();
    Object.entries(form).forEach(([k, v]) => data.append(k, Array.isArray(v) ? v.join(',') : v));
    photos.forEach(p => data.append('photos', p));
 
    try {
      const res = await axios.post('/community/submit', data);
      setStatus('success');
      setMessage(res.data.message);
    } catch (e) {
      setStatus('error');
      setMessage(e.response?.data?.error || 'Something went wrong.');
    }
  };
 
  if (status === 'success') return (
    <div className="p-6 bg-green-50 rounded-2xl text-center">
      <p className="text-2xl mb-2">🎉</p>
      <p className="font-bold text-green-800">Spot submitted!</p>
      <p className="text-sm text-green-700 mt-1">{message}</p>
    </div>
  );
 
  return (
    <div className="p-4 max-w-lg mx-auto">
      <h2 className="text-xl font-bold mb-4">Add a local spot</h2>
 
      <input placeholder="Spot name *" value={form.spot_name}
        onChange={e => setForm(f => ({...f, spot_name: e.target.value}))}
        className="w-full border rounded-xl px-3 py-2 mb-3 text-sm" />
 
      <select value={form.spot_type}
        onChange={e => setForm(f => ({...f, spot_type: e.target.value}))}
        className="w-full border rounded-xl px-3 py-2 mb-3 text-sm">
        <option value="">Type of spot *</option>
        {SPOT_TYPES.map(t => <option key={t}>{t}</option>)}
      </select>
 
      <input placeholder="Area / Locality *" value={form.area}
        onChange={e => setForm(f => ({...f, area: e.target.value}))}
        className="w-full border rounded-xl px-3 py-2 mb-3 text-sm" />
 
      <textarea placeholder="Tell us about this place (what makes it special) *"
        value={form.description} rows={3}
        onChange={e => setForm(f => ({...f, description: e.target.value}))}
        className="w-full border rounded-xl px-3 py-2 mb-3 text-sm" />
 
      <input placeholder="Must-try dish name *" value={form.must_try_dish}
        onChange={e => setForm(f => ({...f, must_try_dish: e.target.value}))}
        className="w-full border rounded-xl px-3 py-2 mb-3 text-sm" />
 
      <textarea placeholder="How would you describe how that dish tastes? *"
        value={form.dish_description} rows={2}
        onChange={e => setForm(f => ({...f, dish_description: e.target.value}))}
        className="w-full border rounded-xl px-3 py-2 mb-3 text-sm" />
 
      <input placeholder="Approx price per person (₹) *" type="number"
        value={form.approx_price}
        onChange={e => setForm(f => ({...f, approx_price: e.target.value}))}
        className="w-full border rounded-xl px-3 py-2 mb-3 text-sm" />
 
      {/* Taste tags */}
      <p className="text-sm font-medium mb-2">Taste tags</p>
      <div className="flex flex-wrap gap-2 mb-4">
        {TASTE_TAGS.map(tag => (
          <button key={tag} onClick={() => toggleTag(tag)}
            className={`text-xs px-3 py-1 rounded-full border transition-colors ${
              form.taste_tags.includes(tag)
                ? 'bg-orange-500 text-white border-orange-500'
                : 'bg-white text-gray-600 border-gray-300'
            }`}>
            {tag}
          </button>
        ))}
      </div>
 
      <input placeholder="Zomato link (optional)" value={form.zomato_url}
        onChange={e => setForm(f => ({...f, zomato_url: e.target.value}))}
        className="w-full border rounded-xl px-3 py-2 mb-3 text-sm" />
 
      <PhotoUploader onPhotosChange={setPhotos} />
 
      {message && <p className="text-sm text-red-600 mb-3">{message}</p>}
 
      <button onClick={handleSubmit} disabled={status === 'loading'}
        className="w-full bg-orange-500 text-white rounded-xl py-3 font-semibold mt-2">
        {status === 'loading' ? 'Submitting...' : 'Submit spot'}
      </button>
    </div>
  );
}
